## 程序分析

* 【顯示中文字`OLED_DISPLAY_16x16(u8 x, u8 y, u16 w)`】
```
u8 j,t,c=0;
for(t=0;t<2;t++){                                        //顯示中文字需要占用2頁，因此每頁需要分別寫入
  I2C_SAND_BYTE(OLED0561_ADD,COM,0xb0+x);                //頁地址(0xB0到0xB7)
  I2C_SAND_BYTE(OLED0561_ADD,COM,y/16+0x10);             //起始列地址的高4位
  I2C_SAND_BYTE(OLED0561_ADD,COM,y%16);                  //起始列地址的低4位
  for(j=0;j<16;j++){                                     //寫入指定頁地址的16列的數據
    I2C_SAND_BYTE(OLED0561_ADD,DAT,GB_16[(w*32)+c]);
    c++;                                                 //下一列需要的十六進制數據
  }
  x++;                                                   //下一個頁地址
}
I2C_SAND_BYTE(OLED0561_ADD,COM,0xAF);                    //開顯示
```


* 【顯示全螢幕圖片`OLED_DISPLAY_PIC1`】
```
u8 m,i;
for(m=0;m<8;m++){                                     //8頁(全螢幕)分別寫入
  I2C_SAND_BYTE(OLED0561_ADD,COM,0xb0+m);             //指定寫入頁地址
  I2C_SAND_BYTE(OLED0561_ADD,COM,0x10);               //起始列地址的高4位
  I2C_SAND_BYTE(OLED0561_ADD,COM,0x02);               //起始列地址的低4位，從最左邊(0x02)開始顯示
  for(i=0;i<128;i++){                                 //每一頁寫入128(整列)次數據
    I2C_SAND_BYTE(OLED0561_ADD,DAT,PIC1[i+m*128]);    //寫入指定頁地址的列數據
  }
}
```
